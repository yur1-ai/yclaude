Create `web/src/store/useThemeStore.ts`:

```typescript
import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';

export type Theme = 'light' | 'dark' | 'system';

interface ThemeState {
  theme: Theme;
  setTheme: (theme: Theme) => void;
}

function applyTheme(theme: Theme): void {
  const root = document.documentElement;
  const isDark =
    theme === 'dark' ||
    (theme === 'system' && window.matchMedia('(prefers-color-scheme: dark)').matches);
  root.classList.toggle('dark', isDark);
}

export const useThemeStore = create<ThemeState>()(
  persist(
    (set) => ({
      theme: 'system' as Theme,
      setTheme: (theme) => {
        applyTheme(theme);
        set({ theme });
      },
    }),
    {
      name: 'yclaude-theme',
      storage: createJSONStorage(() => localStorage),
      onRehydrateStorage: () => (state) => {
        if (state) applyTheme(state.theme);
      },
    }
  )
);
```

Also add a `useEffect` listener for system preference changes when theme is 'system' — add this inside `onRehydrateStorage` or as a module-level side effect that subscribes to `window.matchMedia('(prefers-color-scheme: dark)').addEventListener('change', ...)` and re-calls `applyTheme(getState().theme)` only when stored theme is 'system'.

Then update `web/index.html` — add the FOUC-prevention inline script inside `<head>` BEFORE the closing `</head>` tag. The script reads the Zustand persist storage shape (`{ state: { theme }, version }`), NOT the raw string. Wrap everything in try/catch. The localStorage key is `'yclaude-theme'`:

```html
<script>
  (function () {
    try {
      var raw = localStorage.getItem('yclaude-theme');
      var theme = raw ? JSON.parse(raw).state.theme : 'system';
      var prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
      if (theme === 'dark' || (theme === 'system' && prefersDark)) {
        document.documentElement.classList.add('dark');
      }
    } catch (e) {}
  })();
</script>
```

Critical: The FOUC script key name `'yclaude-theme'` MUST match the Zustand persist `name` field exactly. Any mismatch causes flash on reload.
